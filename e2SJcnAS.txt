local LocalUI = loadstring(game:HttpGet("https://pastebin.com/raw/PnQqq248"))()

-- ⛩️ แท็บหลัก
local Tab1 = CreateTab("หลัก", "⛩️")

_G.AutoFarm = false -- ตั้งค่าเริ่มต้นเป็นปิด

CreateToggled(Tab1, "ฟาม⭐", function(state) -- เพิ่ม 'state' เพื่อรับค่าจากปุ่ม
    _G.AutoFarm = state -- กำหนดค่าตามปุ่มที่กด
    
    if _G.AutoFarm then
        print("เริ่มฟาม: ON")
        
        -- ส่วนที่ 1: การเคลื่อนที่ (ลอย)
        task.spawn(function()
            local TweenService = game:GetService("TweenService")
            local RunService = game:GetService("RunService")
            local player = game.Players.LocalPlayer
            
            local targetPos = Vector3.new(-252, 3, 599)
            local moveSpeed = 50
            local isNoclipping = false

            -- ระบบ Noclip (ทำงานเฉพาะตอนเปิดฟาร์ม)
            local noclipConn
            noclipConn = RunService.Stepped:Connect(function()
                if _G.AutoFarm and isNoclipping then
                    local character = player.Character
                    if character then
                        for _, part in pairs(character:GetDescendants()) do
                            if part:IsA("BasePart") then part.CanCollide = false end
                        end
                    end
                else
                    if not _G.AutoFarm then noclipConn:Disconnect() end
                end
            end)

            while _G.AutoFarm do
                local character = player.Character
                if character and character:FindFirstChild("HumanoidRootPart") then
                    local hrp = character.HumanoidRootPart
                    local distance = (hrp.Position - targetPos).Magnitude
                    
                    if distance > 2 then
                        isNoclipping = true
                        local duration = distance / moveSpeed
                        local tween = TweenService:Create(hrp, TweenInfo.new(duration, Enum.EasingStyle.Linear), {CFrame = CFrame.new(targetPos)})
                        tween:Play()
                        tween.Completed:Wait()
                        isNoclipping = false
                    end
                end
                task.wait(3) -- ลอยมาเช็คทุก 3 วิ
            end
        end)

        -- ส่วนที่ 2: การกดปุ่ม E ค้าง
        task.spawn(function()
            while _G.AutoFarm do
                local player = game.Players.LocalPlayer
                local character = player.Character
                if character and character:FindFirstChild("HumanoidRootPart") then
                    local hrp = character.HumanoidRootPart
                    local targetPos = Vector3.new(-252, 3, 599)

                    if (hrp.Position - targetPos).Magnitude < 15 then
                        for _, prompt in pairs(workspace:GetDescendants()) do
                            if prompt:IsA("ProximityPrompt") then
                                local promptParent = prompt.Parent
                                if promptParent:IsA("BasePart") and (hrp.Position - promptParent.Position).Magnitude <= 1000 then
                                    prompt:InputHoldBegin()
                                    task.wait(prompt.HoldDuration + 0.2)
                                    prompt:InputHoldEnd()
                                end
                            end
                        end
                    end
                end
                task.wait(5.7) -- วนลูปกดทุก 5.7 วิ
            end
        end)
    else
        print("หยุดฟาม: OFF")
    end
end)
